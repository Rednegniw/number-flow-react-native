A space-themed countdown with dramatic Mars imagery, Orbitron typography, and large animated digits ticking toward touchdown.

```LiveExample
import { useEffect, useRef, useState } from 'react';
import { View, Text, Image } from 'react-native';
import { TimeFlow } from 'number-flow-react-native';

const MARS_IMAGE =
  'https://images.unsplash.com/photo-1630694093867-4b947d812bf0?w=800&q=80&auto=format';

const BG = '#050510';

const INITIAL_HOURS = 19;
const INITIAL_MINUTES = 23;
const INITIAL_SECONDS = 45;
const INITIAL_TOTAL =
  INITIAL_HOURS * 3600 + INITIAL_MINUTES * 60 + INITIAL_SECONDS;

function useFontsReady() {
  const [ready, setReady] = useState(false);

  useEffect(() => {
    document.fonts.ready.then(() => setReady(true));
  }, []);

  return ready;
}

export default function MarsCountdown() {
  const fontsReady = useFontsReady();
  const [remaining, setRemaining] = useState(INITIAL_TOTAL);
  const intervalRef = useRef<ReturnType<typeof setInterval> | null>(null);

  useEffect(() => {
    intervalRef.current = setInterval(() => {
      setRemaining(prev => Math.max(0, prev - 1));
    }, 1000);

    return () => {
      if (intervalRef.current) clearInterval(intervalRef.current);
    };
  }, []);

  const hours = Math.floor(remaining / 3600);
  const minutes = Math.floor((remaining % 3600) / 60);
  const seconds = remaining % 60;

  return (
    <View style={{ minHeight: '100vh', backgroundColor: BG }}>
      {/* Mars background — edge-to-edge, looming */}
      <Image
        source={{ uri: MARS_IMAGE }}
        resizeMode="cover"
        style={{
          position: 'absolute',
          top: 0,
          left: 0,
          right: 0,
          height: '75vh',
        }}
      />

      {/* Gradient fade from image to background */}
      <View
        style={{
          position: 'absolute',
          left: 0,
          right: 0,
          top: '50vh',
          height: '30vh',
          background: `linear-gradient(to bottom, transparent, ${BG})`,
        }}
      />

      {/* Content anchored to bottom */}
      <View style={{
        minHeight: '100vh',
        justifyContent: 'flex-end',
        alignItems: 'center',
        paddingBottom: 100,
      }}>
        {/* Header */}
        <Text style={{
          fontFamily: 'Orbitron',
          fontSize: 11,
          fontWeight: '500',
          color: 'rgba(255, 255, 255, 0.5)',
          letterSpacing: 3,
          textTransform: 'uppercase',
          marginBottom: 12,
        }}>
          Landing on Mars in
        </Text>

        {/* Timer box — wait for fonts so glyph measurement is accurate */}
        <View style={{
          borderWidth: 1,
          borderColor: 'rgba(255, 255, 255, 0.15)',
          paddingHorizontal: 24,
          paddingVertical: 14,
          overflow: 'visible',
        }}>
          {fontsReady && (
            <TimeFlow
              hours={hours}
              minutes={minutes}
              seconds={seconds}
              trend={-1}
              style={{
                fontSize: 40,
                fontFamily: 'Orbitron',
                fontWeight: '700',
                fontVariant: ['tabular-nums'],
                color: '#ffffff',
                letterSpacing: 2,
              }}
            />
          )}
        </View>
      </View>
    </View>
  );
}
```


<Callout type="info">
This demo uses a CSS `linear-gradient` for the image-to-background fade. Since `LiveExample` demos run as React Native Web in the browser, CSS properties like `background` work directly in the `style` prop. In a real React Native app, you'd use `expo-linear-gradient` or `react-native-linear-gradient` instead.
</Callout>
