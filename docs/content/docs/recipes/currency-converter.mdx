A minimal currency converter with large animated values, locale-aware formatting, and a clean card-based layout. Watch the decimal separators, currency symbols, and digit grouping animate when switching between currencies.

```SnackPlayer name=Currency%20Converter dependencies=number-flow-react-native,react-native-reanimated
import { useState } from 'react';
import { View, Text, Pressable, Platform } from 'react-native';
import { NumberFlow } from 'number-flow-react-native';

const amounts = [50, 100, 500, 1000, 5000];

export default function CurrencyConverter() {

function CurrencyPicker({
  currencies,
  selectedIndex,
  onChange,
  dark = false,
}: {
  currencies: typeof currencies;
  selectedIndex: number;
  onChange: (index: number) => void;
  dark?: boolean;
}) {
  return (
    <View style={{ flexDirection: 'row', gap: 6 }}>
      {currencies.map((c, i) => {
        const isActive = i === selectedIndex;
        const activeBg = dark ? '#292524' : '#f5f5f4';
        const activeBorder = dark ? '#78716c' : '#d6d3d1';

        return (
          <Pressable
            key={c.code}
            onPress={() => onChange(i)}
            style={{
              paddingHorizontal: 12,
              paddingVertical: 6,
              borderRadius: 8,
              backgroundColor: isActive ? activeBg : 'transparent',
              borderWidth: 1,
              borderColor: isActive ? activeBorder : 'transparent',
            }}
          >
            <Text style={{
              fontSize: 13,
              fontWeight: isActive ? '700' : '500',
              color: dark
                ? (isActive ? '#fafaf9' : '#57534e')
                : (isActive ? '#1c1917' : '#a8a29e'),
            }}>
              {c.code}
            </Text>
          </Pressable>
        );
      })}
    </View>
  );
}
```
