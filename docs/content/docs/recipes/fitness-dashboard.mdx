A dark fitness dashboard with vibrant metric cards, live-updating values, and the `continuous` odometer effect for accumulating metrics.

```SnackPlayer name=Fitness%20Dashboard dependencies=number-flow-react-native,react-native-reanimated
import { useEffect, useState } from 'react';
import { View, Text, Platform } from 'react-native';
import { NumberFlow } from 'number-flow-react-native';

interface Metric {
  label: string;
  value: number;
  format?: Intl.NumberFormatOptions;
  suffix: string;
  color: string;
  dimColor: string;
  icon: string;
}

export default function FitnessDashboard() {
  const [calories, setCalories] = useState(1247);
  const [distance, setDistance] = useState(5.83);
  const [heartRate, setHeartRate] = useState(72);

  useEffect(() => {
    const interval = setInterval(() => {
      setCalories(v => v + Math.floor(Math.random() * 8) + 1);
      setDistance(v => v + Math.random() * 0.02);
      setHeartRate(v => {
        const delta = Math.floor(Math.random() * 5) - 2;
        return Math.max(55, Math.min(180, v + delta));
      });
    }, 3000);
    return () => clearInterval(interval);
  }, []);

  const metrics: Metric[] = [
    {
      label: 'Active Calories',
      value: calories,
      suffix: ' kcal',
      color: '#f97316',
      dimColor: 'rgba(249, 115, 22, 0.12)',
      icon: 'üî•',
    },
    {
      label: 'Distance',
      value: distance,
      format: { minimumFractionDigits: 2, maximumFractionDigits: 2 },
      suffix: ' km',
      color: '#06b6d4',
      dimColor: 'rgba(6, 182, 212, 0.12)',
      icon: 'üìç',
    },
    {
      label: 'Heart Rate',
      value: heartRate,
      suffix: ' bpm',
      color: '#f43f5e',
      dimColor: 'rgba(244, 63, 94, 0.12)',
      icon: '‚ô•',
    },
  ];

  return (
    <View style={{
      flex: 1,
      justifyContent: 'center',
      backgroundColor: '#09090b',
      padding: 20,
      gap: 12,
    }}>
      {/* Header */}
      <View style={{ paddingHorizontal: 4, marginBottom: 8 }}>
        <Text style={{
          fontSize: 13,
          fontWeight: '600',
          color: '#52525b',
          letterSpacing: 3,
          textTransform: 'uppercase',
        }}>
          Activity
        </Text>
        <Text style={{
          fontSize: 28,
          fontWeight: '800',
          color: '#fafafa',
          fontFamily: Platform.OS === 'ios' ? 'SF Pro Rounded' : 'sans-serif',
          letterSpacing: -0.5,
          marginTop: 2,
        }}>
          Today
        </Text>
      </View>

      {/* Metric cards */}
      {metrics.map(metric => (
        <View
          key={metric.label}
          style={{
            flexDirection: 'row',
            alignItems: 'center',
            padding: 20,
            borderRadius: 20,
            backgroundColor: '#18181b',
            borderWidth: 1,
            borderColor: '#27272a',
            gap: 16,
          }}
        >
          {/* Icon badge */}
          <View style={{
            width: 48,
            height: 48,
            borderRadius: 14,
            backgroundColor: metric.dimColor,
            alignItems: 'center',
            justifyContent: 'center',
          }}>
            <Text style={{ fontSize: 22 }}>{metric.icon}</Text>
          </View>

          {/* Value */}
          <View style={{ flex: 1, gap: 2 }}>
            <Text style={{
              fontSize: 12,
              fontWeight: '600',
              color: '#52525b',
              letterSpacing: 1,
              textTransform: 'uppercase',
            }}>
              {metric.label}
            </Text>
            <NumberFlow
              value={metric.value}
              format={metric.format}
              suffix={metric.suffix}
              continuous
              style={{
                fontSize: 30,
                fontFamily: Platform.OS === 'ios' ? 'SF Pro Rounded' : 'sans-serif',
                fontWeight: '700',
                color: metric.color,
              }}
            />
          </View>
        </View>
      ))}
    </View>
  );
}
```

### Features demonstrated

- **Rounded font** ‚Äî `'SF Pro Rounded'` gives large numbers a soft, fitness-app feel typical of Apple Health and similar apps
- **Continuous odometer** ‚Äî the `continuous` prop creates cascading digit rolls as calories and distance accumulate
- **Colored accent badges** ‚Äî each metric has a translucent icon background matching its accent color
- **Dark card layout** ‚Äî `#18181b` cards on `#09090b` background with subtle `#27272a` borders

<Callout type="info">
Each `NumberFlow` instance manages its own animation state independently. When calories, distance, and heart rate update simultaneously (every 3 seconds in this example), all three animate concurrently without interfering with each other. The fixed slot pool means no component remounting even when digit counts change (e.g. calories going from `999` to `1000`).
</Callout>
