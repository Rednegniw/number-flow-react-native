---
title: Trend Control
description: Control digit spin direction with static values or dynamic functions
---

The `trend` prop controls which direction digits spin during transitions. You can pass a static value or a function for dynamic control.

```SnackPlayer name=Trend%20Control dependencies=number-flow-react-native,react-native-reanimated
import { useState } from 'react';
import { View, Button, Text } from 'react-native';
import { NumberFlow } from 'number-flow-react-native';
import type { Trend } from 'number-flow-react-native';

export default function TrendControl() {
  const [value, setValue] = useState(50);
  const [trend, setTrend] = useState<Trend | undefined>(undefined);

  return (
    <View style={{ alignItems: 'center', gap: 24, padding: 32 }}>
      <NumberFlow
        value={value}
        trend={trend}
        style={{ fontSize: 48, fontFamily: 'System', color: '#000' }}
      />

      {/* Value controls */}
      <View style={{ flexDirection: 'row', gap: 8 }}>
        <Button title="-10" onPress={() => setValue(v => v - 10)} />
        <Button title="+10" onPress={() => setValue(v => v + 10)} />
        <Button title="Random" onPress={() => setValue(Math.floor(Math.random() * 100))} />
      </View>

      {/* Trend controls */}
      <View style={{ gap: 8 }}>
        <Text style={{ fontSize: 14, color: '#666', textAlign: 'center' }}>Trend Direction</Text>
        <View style={{ flexDirection: 'row', gap: 8 }}>
          <Button title="Auto" onPress={() => setTrend(undefined)} color={trend === undefined ? '#007AFF' : '#999'} />
          <Button title="Up (1)" onPress={() => setTrend(1)} color={trend === 1 ? '#007AFF' : '#999'} />
          <Button title="Down (-1)" onPress={() => setTrend(-1)} color={trend === -1 ? '#007AFF' : '#999'} />
          <Button title="Shortest (0)" onPress={() => setTrend(0)} color={trend === 0 ? '#007AFF' : '#999'} />
        </View>
      </View>
    </View>
  );
}
```

### Trend values

| Value | Behavior |
|---|---|
| `undefined` (default) | Auto-detects: increasing values spin up, decreasing spin down |
| `1` | Always spin upward |
| `-1` | Always spin downward |
| `0` | Each digit takes the shortest path (e.g. 9→1 goes up through 0 instead of rolling down through 8,7,6...) |

### Dynamic trend function

You can also pass a function that receives the previous and next values:

```tsx
<NumberFlow
  value={price}
  trend={(prev, next) => {
    if (next > prev) return 1;  // Price up → spin up
    if (next < prev) return -1; // Price down → spin down
    return 0;                    // Same → shortest path
  }}
  style={{ fontSize: 36, fontFamily: 'System', color: '#000' }}
/>
```
