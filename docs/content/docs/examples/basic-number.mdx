The simplest `NumberFlow` usage â€” a number that rolls digit-by-digit when the value changes.

```LiveExample
import { useState } from 'react';
import { View } from 'react-native';
import { NumberFlow } from 'number-flow-react-native';
import { ExampleLayout, Button } from './ExampleUI';

export default function BasicNumber() {
  const [value, setValue] = useState(42);

  return (
    <ExampleLayout>
      <NumberFlow
        value={value}
        style={{ fontSize: 48, fontWeight: '600', color: '#fff' }}
      />
      <View style={{ flexDirection: 'row', gap: 8, justifyContent: 'center' }}>
        <Button title="-10" onPress={() => setValue(v => v - 10)} />
        <Button title="-1" onPress={() => setValue(v => v - 1)} />
        <Button title="+1" onPress={() => setValue(v => v + 1)} />
        <Button title="+10" onPress={() => setValue(v => v + 10)} />
      </View>
    </ExampleLayout>
  );
}
```
